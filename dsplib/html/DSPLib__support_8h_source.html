202  int16_t incr,
203  uintptr_t mask)
204 {
205  uintptr_t base = (uintptr_t)__circular_mask(ptr, mask);

260 static inline void msp_checkStatus(msp_status status)
261 {
262 #ifndef MSP_DISABLE_DIAGNOSTICS
263  switch (status) {
264  case MSP_SUCCESS:
265  break;
266  case MSP_SIZE_ERROR:
267  /* Invalid size error, loop forever. */
268  while(true) __no_operation();
269  case MSP_SHIFT_SIZE_ERROR:
270  /* Invalid shift size error, loop forever. */
271  while(true) __no_operation();
272  case MSP_TABLE_SIZE_ERROR:
273  /* Invalid table size error, loop forever. */
274  while(true) __no_operation();
275  case MSP_LEA_BUSY:
276  /* LEA is busy and cannot be invoked, loop forever. */
277  while(true) __no_operation();
278  case MSP_LEA_INVALID_ADDRESS:
279  /* Address is not within LEA RAM section, loop forever. */
280  while(true) __no_operation();
281  case MSP_LEA_OUT_OF_RANGE:
282  /* Command resulted in out of range memory access, loop forever. */
283  while(true) __no_operation();
284  case MSP_LEA_SCALAR_INCONSISTENCY:
285  /* LEA scalar data inconsistency error, loop forever. */
286  while(true) __no_operation();
287  case MSP_LEA_COMMAND_OVERFLOW:
288  /* LEA command overflow error, loop forever. */
289  while(true) __no_operation();
290  case MSP_LEA_INCORRECT_REVISION:
291  /* LEA incorrect revision, loop forever. */
292  while(true) __no_operation();
293  default:
294  /* Unknown error, loop forever. */
295  while(true) __no_operation();
296  }
297 #endif //MSP_DISABLE_DIAGNOSTICS
298 }

334 static inline void msp_benchmarkStart(uint16_t baseAddress, uint8_t res)
335 {
336  static const uint8_t TAx_ID[8] = {
337  0x00, 0x40, 0x40, 0x80, 0x80, 0x80, 0x80, 0xC0
338  };
339  HWREG16(baseAddress) = 0x0004;
340  HWREG16(baseAddress+BENCHMARK_TAxEX0) = res < 7 ? 0 : ((0x07&(res >> 3))-1);
341  HWREG16(baseAddress) = 0x0220 + TAx_ID[res > 7 ? 7 : ((res-1)&7)];
342 }

344 //******************************************************************************
345 //
353 //
354 //******************************************************************************
355 static inline uint32_t msp_benchmarkStop(uint16_t baseAddress)
356 {
357  static const uint8_t TAx_IDEX_SCALE[8] = {
358  0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08
359  };
360  uint16_t cycles = HWREG16(baseAddress+BENCHMARK_TAxR);
361  uint16_t shift = (HWREG16(baseAddress) & 0x00C0) >> 6;
362  uint16_t scale = TAx_IDEX_SCALE[HWREG16(baseAddress+BENCHMARK_TAxEX0) & 7];
363  HWREG16(baseAddress) = 0;
364  return (scale*((uint32_t)cycles << shift));
365 }
