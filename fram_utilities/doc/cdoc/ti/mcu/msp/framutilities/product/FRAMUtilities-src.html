1    @Template("./FRAMUtilities.xdt")
     2    metaonly module FRAMUtilities
     3    {
     4        /*!
     5         *  ======== BASENAME ========
     6         *  The base name of the pre-built DriverLib library
     7         *  
     8         *  This name controls the name of a pre-built DriverLib library.  For
     9         *  example, if BASENAME is set to "driverlib" the name of the library
    10         *  will be driverlib.lib in the ./src/driverlib folder of your project.
    11         *  
    12         *  @_nodoc
    13         */
    14        const String BASENAME = "framutilities";
    15    
    16        /*!
    17         *  ======== getFamilyPath ========
    18         *  Given a device name, return the path to the device family's sources
    19         *
    20         *  This function returns a path relative to the DriverLib source
    21         *  repository containing the DriverLib sources.  For example,
    22         *  if `deviceName` is `"MSP430FR5969"`, this function returns
    23         *  `"driverlib/MSP430FR5xx_6xx"`.
    24         *
    25         *  The string returned by this function can be used with `getSrcRepo`
    26         *  to obtain the absolute path to the sources for the DriverLib family
    27         *  that supports the device `deviceName`.
    28         *  
    29         *  @see #getSrcRepo
    30         */
    31        String getFamilyPath(String deviceName);
    32    
    33        /*!
    34         *  ======== getLibPath ========
    35         *  Returns the path to a pre-built DriverLib library
    36         *  
    37         *  This function is called by the `getLibs()` method provided by the
    38         *  package containing the DriverLib module.  This `getLibs()` method
    39         *  is implemented in the `package.xs` file and is used to provide the
    40         *  path to a pre-built DriverLib library for any project that uses
    41         *  DriverLib.
    42         *  
    43         *  @_nodoc
    44         */
    45        String getLibPath();
    46    
    47        /*!
    48         *  ======== getSrcRepo ========
    49         *  Given a family path, return the path to it's source repository
    50         *
    51         *  This function returns an absolute path to the repository
    52         *  containing the DriverLib sources for the specified device family.
    53         *
    54         *  @see #getFamilyPath
    55         */
    56        String getSrcRepo(String familyPath);
    57    
    58        /*!
    59         *  ======== outputDir ========
    60         *  The output directory for all generated sources
    61         *
    62         *  The default output directory is the `src/` sub-directory of the
    63         *  directory containing the configuration script used to specify
    64         *  the peripherals for which to generate code.
    65         *
    66         *  This parameter should only be changed when running the
    67         *  configuration tool in a command line environment.  Integrated
    68         *  Development Environments (IDEs) often require that generated sources
    69         *  be placed in specific directories.  Changes to this parameter often
    70         *  require corresponding changes to the IDE.
    71         *
    72         *  @_nodoc
    73         */
    74        config String outputDir;
    75    }
