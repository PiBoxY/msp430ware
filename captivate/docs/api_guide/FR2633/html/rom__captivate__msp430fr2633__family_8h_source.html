18 //*****************************************************************************
19 //
20 // CapTIvate ROM API Definitions
21 // - ROM_CAPTIVATE_APITABLE defines the beginning of the ROM section.
22 // - ROM_CAPTIVATETABLE is the ROM defined indirect address of the jump table.
23 // - ROM_CAPTIVATE_FXNTABLE is the direct address of the jump table.
24 // - ROM_CAPTIVATE_JUMPTABLE is the table address to use, and should be
25 // selected as ROM_CAPTIVATETABLE or ROM_CAPTIVATE_FXNTABLE.
26 //
27 //*****************************************************************************
28 #define ROM_CAPTIVATE_APITABLE ((uint16_t *)0x4000)
29 #define ROM_CAPTIVATE_VERSIONL (ROM_CAPTIVATE_APITABLE[0])
30 #define ROM_CAPTIVATE_VERSIONH (ROM_CAPTIVATE_APITABLE[1])
31 #define ROM_CAPTIVATETABLE ((uint16_t *)(ROM_CAPTIVATE_APITABLE[2]))
32 #define ROM_CAPTIVATE_FXNTABLE ((uint16_t *)0x4006)
33 #define ROM_CAPTIVATE_JUMPTABLE (ROM_CAPTIVATE_FXNTABLE)
