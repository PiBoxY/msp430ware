93 {

95 
96  // check if logging should be disabled
97  for(i=1;i<argc;i++)
98  {

100  {
101  inst_state |= LOG_OFF;

103  }
104 
105  if ((~inst_state & LOG_OFF) && ((lfile = fopen("Log/log.txt", "a")) == NULL))
106  {
107  printf("\n");

110 #ifdef WIN32

292 int EvaluateOptions(int argc, char* argv[], sDefOptions_t* sOpt, sRTE_t* RTE)
293 {
294  int RetState;
295  unsigned long lCountAddOpt;
296  unsigned long i;
297  char cOpt;
298  char* mk_upper = NULL;
299  unsigned short OptFlags;
300 
301  RetState = 0;
302  OptFlags = 0;
303  lCountAddOpt= 0;
304 
305  Print_N_Log(REPORT_MSG,"* Evaluating triggers...");
306 
307  // get parameter
308  while ((cOpt = getopt(argc, ( char**)argv, g_cAllOpts, RTE)) != EOF )
309  {
310  // choose code for determined parameter
311  switch (cOpt)
312  {
313  case 'n': // device name
314  case 'N':
315  sOpt->pcDeviceName = optarg;
316  if((string(sOpt->pcDeviceName) == "MSP430L092") ||
317  (string(sOpt->pcDeviceName) == "MSP430C092") ||
318  (string(sOpt->pcDeviceName).find("RF430") != string::npos))
319  {
320  inst_state |= MODE_SELECT;
321  g_Def.deviceAC = 0xA55AA55A;
322  }
323  if(string(sOpt->pcDeviceName).find("MSP432") != string::npos)
324  {
325  inst_state |= TARGETARCH_432;
326  }
327  break;
328  case 'o': // device mode
329  case 'O':
330  sOpt->pcDeviceMode = optarg;
331  inst_state |= MODE_SELECT;
332  break;
333  case 'p': // password
334  case 'P':
335  sOpt->pcPassword = optarg;
336  OptFlags |= OPT_PASSWORD;
337  break;
338  case 'l': // password length
339  case 'L': // OBSOLETE
340  break;
341  case 'i': // interface
342  case 'I':
343  sOpt->pcInterface = optarg;
344  OptFlags |= OPT_INTERFACE;
345  break;
346  case 'j': // InterfaceSpeed
347  case 'J':
348  sOpt->pcFETspeed = optarg;
349  OptFlags |= OPT_FETSPEED;
350  break;
351  case 'm': // connection mode
352  case 'M': // OBSOLETE
353  break;
354  case 'r': // read specifications
355  case 'R':
356  sOpt->pcReadSpecs = optarg;
357  OptFlags |= OPT_READSPECS;
358  break;
359  case 'w': // input file
360  case 'W':
361  sOpt->pcFile = optarg;
362  OptFlags |= OPT_INFILE;
363  break;
364  case 'a': // no reset during start-up
365  case 'A':
366  inst_state |= NO_RESET;
367  break;
368  case 'b': // BSL memory access
369  case 'B':
370  inst_state |= WRITE_BSL;
371  break;
372  case 'u': // locked flash memory access
373  case 'U':
374  inst_state |= UNLOCK_FLASH;
375  break;
376  case 'e': // erasetype
377  case 'E':
378  sOpt->pcEraseType = optarg;
379  if(sOpt->pcFile == NULL && strcmp(sOpt->pcEraseType, "NO_ERASE"))
380  {
381  inst_state |= ERASE_NO_WRITE;
382  }
383  break;
384  case 'f': // device secure
385  case 'F':
386  inst_state |= BLOW_FUSE;
387  break;
388  case 'v': // verify
389  case 'V':
390  if(optarg)
391  {
392  sOpt->pcVerify = optarg;
393  }
394  else
395  {
396  sOpt->pcVerify = g_Def.defTRUE;
397  }
398  break;
399  case 'z': // exit specs
400  case 'Z':
401  sOpt->pcExitSpecs = optarg;
402  OptFlags |= OPT_EXITSPECS;
403  break;
404  case 'g': // disable log
405  case 'G': // handled in EnableLog()
406  inst_state |= LOG_OFF;
407  break;
408  case 'q': // quiet mode
409  case 'Q':
410  inst_state |= QUIET_MODE;
411  break;
412  case 'H': // show help on the display
413  case 'h':
414  case '?':
415  Print_N_Log(REPORT_MSG,"");
416  return(RetState = 1);
417  break;
418  case 'x': // show exit specs
419  case 'X':
420  Print_N_Log(REPORT_MSG,"");
421  return(RetState = 2);
422  break;
423  case 'd': // breakpoints
424  case 'D':
425  sOpt->pcBreakpoints = optarg;
426  OptFlags |= OPT_BPSPECS;
427  break;
428  case 't': // breakpoint timeout
429  case 'T':
430  sOpt->pcTimeout = optarg;
431  break;
432  case 's': // suppress FET firmware update prompt
433  case 'S':
434  inst_state |= SKIP_UDPROMPT;
435  break;
436  }
437  }
438 
439  // Special handling for devices with ROM development mode
440  if(inst_state & MODE_SELECT)
441  {
442 
443  char device_mode_entry; // for storing device mode if entered during runtime
444 
445  if ((sOpt->pcDeviceMode == NULL) || // if device mode is not specified
446  (((strchr("Ll", *(sOpt->pcDeviceMode)) == NULL) && // if entered character is...
447  (strchr("Cc", *(sOpt->pcDeviceMode)) == NULL)) || // ...invalid
448  (strlen(sOpt->pcDeviceMode) > 1))) // if entered mode has more than one character
449  {
450  if (sOpt->pcDeviceMode == NULL) // if device mode is not specified
451  {
452  Print_N_Log(ERR_MSG,"- The device you are trying to access can\n");
453  Print_N_Log(ERR_MSG,"- operate in multiple modes but no mode has\n");
454  Print_N_Log(ERR_MSG,"- been specified. Please use trigger -o. \n");
455  }
456  else if (((strchr("Ll", *(sOpt->pcDeviceMode)) == NULL) && // if entered character is...
457  (strchr("Cc", *(sOpt->pcDeviceMode)) == NULL)) || // ...invalid
458  (strlen(sOpt->pcDeviceMode) > 1)) // if entered mode has more than one character
459  {
460  Print_N_Log(ERR_MSG,"* Invalid argument for -o switch\n");
461  }
462  Print_N_Log(ERR_MSG,"- Please enter L for normal mode, C for\n"); // prompt user for device mode
463  Print_N_Log(ERR_MSG,"- ROM development mode or enter n to abort:\n");
464  do
465  {
466  cin >> device_mode_entry;
467  cin.ignore();
468  Print_N_Log(LOG_MSG, "%c\n", device_mode_entry);
469  }
470  while(strchr("lLcCn", device_mode_entry) == NULL);
471 
472  if (device_mode_entry == 'n')
473  {
474  return(RetState = 1); // perform exit on abort
475  }
476  else
477  {
478  sOpt->pcDeviceMode = &device_mode_entry; // update device mode
479  }
480  }
481 
482  if(strchr("Ll", *(sOpt->pcDeviceMode)) != NULL)
483  {
484  g_Def.deviceAC = 0xA55AA55A;
485  if(string(sOpt->pcDeviceName).find("RF430") == string::npos)
486  {
487  inst_state |= WRITE_EXT_MEM;
488  }
489  }
490  else if (strchr("Cc", *(sOpt->pcDeviceMode)) != NULL)
491  {
492  g_Def.deviceAC = 0x5AA55AA5;
493  }
494  }
495 
496  if(OptFlags & OPT_FETSPEED)
497  {
498  for(mk_upper = sOpt->pcFETspeed; *mk_upper != '\0'; mk_upper++) // converts argument to uppercase
499  *mk_upper = toupper(*mk_upper);
500  string FETspeed(sOpt->pcFETspeed ? sOpt->pcFETspeed : "");
501  if(FETspeed != "MEDIUM" && FETspeed != "FAST" && FETspeed != "SLOW")
502  {
503  Print_N_Log(ERR_MSG,"* Invalid argument for -j switch.\n");
504  Print_N_Log(ERR_MSG,"* Terminating...\n");
505  exit(EXIT_VARIOUS);
506  }
507  }
508 
509  if(OptFlags & OPT_PASSWORD)
510  {
511  string Password(sOpt->pcPassword ? sOpt->pcPassword : "");
512 
513  if(Password.find("0x") == string::npos) // if entered password is not hex
514  {
515  Print_N_Log(ERR_MSG,"* Please enter hex password (beginning with 0x)\n");
516  Print_N_Log(ERR_MSG,"* Terminating...\n");
517  exit(EXIT_VARIOUS);
518  }
519  if ((Password.length() - 2) % 4 != 0) // if entered password is not multiple of 16-bit words
520  {
521  Print_N_Log(ERR_MSG,"* Invalid password length!\n");
522  Print_N_Log(ERR_MSG,"* Please enter a multiple of 16 bit words.\n");
523  Print_N_Log(ERR_MSG,"* Terminating...\n");
524  exit(EXIT_VARIOUS);
525  }
526  else
527  {
528  sOpt->pcPWlen = (Password.length() - 2) / 4;
529  }
530  }
531 
532  if (OptFlags & OPT_INFILE)
533  {
534  // filename must be valid. Check:
535  while(fopen(sOpt->pcFile, "r") == NULL)
536  {
537  Print_N_Log(ERR_MSG,"- The entered filename is invalid.\n");
538  Print_N_Log(ERR_MSG,"-\n");
539  Print_N_Log(ERR_MSG,"- Please reenter the filename or enter n to abort: ");
540 
541  cin.getline(RTE->filename_entry, MAX_PATH_LEN);
542  Print_N_Log(LOG_MSG, "%s\n", RTE->filename_entry);
543 
544  string RTE_FileName(RTE->filename_entry ? RTE->filename_entry : "");
545  if(RTE_FileName == "n" || RTE_FileName == "N")
546  {
547  return(RetState = 1); // perform exit on abort
548  }
549  else
550  {
551  sOpt->pcFile = RTE->filename_entry; // update filename
552  }
553  }
554  }
555 
556  // evaluate exit specs
557  if(OptFlags & OPT_EXITSPECS)
558  {
559  char* exit_spec_start = NULL; // pointers for separating delimited exit specs
560  char* exit_spec_stop = NULL;
561  char exit_spec[MAX_NAME_LEN]; // for storing individual exit specs temporarily for processing
562 
563  exit_spec_start = sOpt->pcExitSpecs + 1;
564  for(exit_spec_stop = exit_spec_start; *exit_spec_stop != '\0'; exit_spec_stop++)
565  {
566  if(strchr(",]", *exit_spec_stop) != NULL)
567  {
568  strncpy(exit_spec, exit_spec_start, exit_spec_stop - exit_spec_start);
569  exit_spec[exit_spec_stop - exit_spec_start] = '\0';
570  if (string(exit_spec) == "VCC")
571  {
572  sOpt->pcVCC = TARGET_VCC;
573  }
574  else if(string(exit_spec).find("=") != std::string::npos)
575  {
576  char remove[] = "VCC= ";
577  string str = string(exit_spec);
578 
579  for(unsigned int i=0; i < strlen(remove); ++i)
580  {
581  str.erase(std::remove(str.begin(), str.end(), remove[i]), str.end());
582  }
583 
584  sOpt->pcVCC = stol(str,nullptr);
585  }
586  else if(string(exit_spec) == "RESET")
587  {
588  inst_state |= RESET_ON_EXIT;
589  }
590  else
591  {
592  Print_N_Log(STD_MSG, "* Exit specification '%s' is invalid and will be ignored.\n", exit_spec);
593  }
594  if(*exit_spec_stop == ']')
595  {
596  break;
597  }
598  exit_spec_start = exit_spec_stop + 1;
599  }
600  }
601  }
602 
603  // evaluate breakpoint specs
604  if(OptFlags & OPT_BPSPECS)
605  {
606  char* bp_specs = NULL;
607  char bp_temp[MAX_NAME_LEN]; //for storing individual breakpoint addresses temporarily
608  int j = 0;
609 
610  bp_specs = sOpt->pcBreakpoints;
611 
612  while(*bp_specs != ']')
613  {
614  bp_specs++; // skip '[' and ','
615  for(bp_specs; (*bp_specs != ',') && (*bp_specs != ']'); bp_specs++)
616  {
617  if(*bp_specs != ' ')
618  {

783

785  argv[optind][0] != '-' || // the current parameter doesn't begin with '-' OR
786  argv[optind][1] == '\0') // the current parameter is a single character
787  {
788  optind = 1; // reset index

790  }

1055 int ReadSpacedArgs(int argc, char* argv[], int* p_optind, unsigned short specs, char* processed)
1056 {
1057  char* optarg; // pointer to first part of argument
1058  int optind = *p_optind; // argument index
1059  int runtime_end; // index to last part of argument
1060 
1061  if (specs & IS_ATTACHED) // if argument is attached to trigger
1062  {
1063  // first part of argument follows trigger
1064  optarg = &argv[optind][2];
1065  }
1066  else
1067  {
1068  // first part of argument is the following input argument (optind++ externally)
1069  optarg = argv[optind];
1070  }
1071 
1072  if (!(specs & IS_FILENAME)) // if argument is of type '[]' specs
1073  {
1074  if (*optarg != '[') // if the argument doesn't begin with '['
1075  {
1076  Print_N_Log(ERR_MSG,"* Error: '[' expected at beginning of\n");
1077  Print_N_Log(ERR_MSG,"* exit_spec list -- z\n");
1078  return('?');
1079  }
1080  }
1081  strcpy(processed, optarg); // copy first argument into processed
1082 
1083  if (!(specs & IS_FILENAME)) // if argument is of type '[]' specs
1084  {
1085  for (runtime_end = optind++; runtime_end < argc; runtime_end++) // look through all following input args
1086  { //optind++ (first part read already)
1087  if (strchr(argv[runtime_end], ']') != NULL ) // until ']' is reached, then:
1088  {
1089  break; // stop counting
1090  }
1091  }
1092  if (runtime_end >= argc) // if ']' is not found
1093  {
1094  Print_N_Log(ERR_MSG,"* Error: ']' expected at end of\n");
1095  Print_N_Log(ERR_MSG,"* exit_spec list -- z\n\n");
1096  return('?');
1097  } // runtime_end indexes last part of arg
1098  }
1099  else // else if argument is of type filename
1100  {
1101  for (runtime_end = optind++; runtime_end < argc; runtime_end++) // search for an extension until found
1102  {
1103  if( ( strstr(argv[runtime_end], ".txt") != NULL ) || ( strstr(argv[runtime_end], ".TXT") != NULL ) ||
1104  ( strstr(argv[runtime_end], ".a43") != NULL ) || ( strstr(argv[runtime_end], ".A43") != NULL ) ||
1105  ( strstr(argv[runtime_end], ".hex") != NULL ) || ( strstr(argv[runtime_end], ".HEX") != NULL ) ||
1106  ( strstr(argv[runtime_end], ".int") != NULL ) || ( strstr(argv[runtime_end], ".INT") != NULL ))
1107  {
1108  break;
1109  }
1110  }
1111  if (runtime_end >= argc) // if not found
1112  {
1113  Print_N_Log(ERR_MSG,"* file extension unknown -- w\n\n");
1114  return('?');
1115  } // runtime_end indexes last part of arg
1116  }
1117 
1118  if (specs & REMOVE_SPACES) // if remove ' ' is set
1119  {
1120  while (optind <= runtime_end) // while unread parts exist
1121  {
1122  strcat(processed, argv[optind++]); // add to processed, removing ' '
1123  }
1124  }
1125  else
1126  {
1127  while (optind <= runtime_end) // while unread parts exist
1128  {
1129  *(strchr(processed, '\0')+1) = '\0'; // add to processed
1130  *strchr(processed, '\0') = ' '; // w/o removing ' '
1131  strcat(processed, argv[optind++]);
1132  }
1133  }
1134 
1135  *p_optind = optind; // update value of external optind
1136  return(0);
1137 }

1146 STATUS_T ParseInputFile(char targetcode[], int32_t* lStartAddress, int32_t* lProgramLength)
1147 {
1148  char* buffer = NULL;
1149  int32_t lSize = 0;
1150  size_t result;
1151 
1152  int16_t start = 0;
1153  int16_t length = 0;
1154 
1155  FILE* lInputFile = NULL;
1156 
1157  if((lInputFile = fopen(targetcode, "r")) == NULL)
1158  {
1159  Print_N_Log(ERR_MSG,"* Failed to open input file!\n");
1160  return STATUS_ERROR;
1161  }
1162  else
1163  {
1164  fseek(lInputFile , 0 , SEEK_END);
1165  lSize = ftell(lInputFile);
1166  rewind(lInputFile);
1167 
1168  buffer = (char*) malloc (sizeof(char)*lSize + 1); // allocate memory
1169  result = fread(buffer, 1, lSize, lInputFile); // copy file into the buffer
1170  buffer[lSize] = '\0';
1171 
1172  fclose(lInputFile);
1173 
1174  std::string tmp(buffer);
1175  free(buffer);
1176 
1177  if((strstr(targetcode, ".txt") != NULL) || (strstr(targetcode, ".TXT") != NULL))
1178  {
1179  std::smatch match;
1180  // regex pattern for TI-txt (captures full memory blocks)
1181  std::regex regex("@(.+)\n([^qQ@]+)");
1182 
1183  std::regex_search(tmp, match, regex);
1184  std::string address = match[1];
1185  std::string code = match[2];
1186 
1187  code.erase(remove(code.begin(), code.end(), ' '), code.end()); //remove whitespaces
1188  code.erase(remove(code.begin(), code.end(), '\n'), code.end()); //and newlines
1189 
1190  char * p;
1191  *lStartAddress = (int32_t)strtol(address.c_str(), &p, 16);
1192  *lProgramLength = (int32_t)(code.size()/2);
1193  }
1194  else if((strstr(targetcode, ".a43") != NULL) || (strstr(targetcode, ".A43") != NULL) ||
1195  (strstr(targetcode, ".hex") != NULL) || (strstr(targetcode, ".HEX") != NULL) ||
1196  (strstr(targetcode, ".int") != NULL) || (strstr(targetcode, ".INT") != NULL))
1197  {
1198  std::smatch match;
1199  // regex pattern for INTEL-hex (captures data from one line only)
1200  std::regex regex(":(\\w\\w)(\\w\\w\\w\\w)(\\w\\w)(\\w*)\\w\\w");
1201 
1202  int32_t next_address = 0;
1203  int32_t bytecount = 0;
1204  int32_t address = 0;
1205  int32_t recordtype = 0;
1206  int32_t prev_recordtype = 0;
1207  int32_t length = 0;
1208 
1209  while(std::regex_search(tmp, match, regex))
1210  {
1211  char * p;
1212  bytecount = strtol(string(match[1]).c_str(), &p, 16);
1213  address = strtol(string(match[2]).c_str(), &p, 16);
1214  recordtype = strtol(string(match[3]).c_str(), &p, 16);
1215  std::string data = match[4];
1216 
1217  length += bytecount;
1218 
1219  if(next_address)
1220  {
1221  if((prev_recordtype == 2) ||
1222  ((prev_recordtype == 4) &&
1223  (recordtype == 0)))
1224  {
1225  *lStartAddress += address;
1226  }
1227  else if(address != next_address)
1228  {
1229  break;
1230  }
1231  }
1232  else
1233  {
1234  switch(recordtype)
1235  {
1236  case 0:
1237  *lStartAddress = address;
1238  case 2:
1239  *lStartAddress = address << 4;
1240  case 4:
1241  *lStartAddress = address << 16;
1242  }
1243  }
1244 
1245  next_address = address + bytecount;
1246  tmp.erase(0, bytecount + 12);
1247  }
1248 
1249  *lProgramLength = length;
1250  }
1251 
1252  return STATUS_OK;
1253  }
