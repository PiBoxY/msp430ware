451 void StdUseCase_CheckFirmware(sDefOptions_t* sOpt)
452 {
453  bool uptodate = true;
454 
455  Print_N_Log(REPORT_MSG,"* Checking firmware compatibility: ");
456 
457  if (lVersion == DLL_FW_MISMATCH || lVersion == V2_TO_V3_UPGRADE_REQUIRED) // if conflict between firmware versions
458  {
459  uptodate = false;
460  Print_N_Log(ERR_MSG,"* The firmware of your FET is outdated.\n"); // prompt user about an update
461 
462  if(inst_state & SKIP_UDPROMPT)
463  {
464  Ret_Initialize = STATUS_ERROR; // convention, set to error first
465 
466  Print_N_Log(REPORT_MSG,"* Skipped firmware update prompt.\n");
467  if(MSP430_FET_FwUpdate(NULL, &MY_MSP430_CallbackFunc, myHandle) != STATUS_OK) // if update fails
468  {
469  inst_state |= ALT_EXIT;
470  StdUseCase_Exit(EXIT_FET_FWUPDATE, 0);
471  }
472  if( (lVersion == V2_TO_V3_UPGRADE_REQUIRED) && (Ret_FWUpdate == STATUS_OK) ) // UIF is now enumerated as CDC device, so reinitialize
473  {
474  Print_N_Log(REPORT_MSG,"* FET has been updated to DLLv3. Reinitializing...\n");
475  Sleep(10000);
476  }
477 
478  Ret_Initialize = MSP430_Initialize(sOpt->pcInterface, &lVersion);
479  if (Ret_Initialize != STATUS_OK)
480  {
481  inst_state |= ALT_EXIT;
482  StdUseCase_Exit(EXIT_INITIALIZE, 0);
483  }
484  uptodate = true;
485  }
486  else
487  {
488  bool updProcRunning = true;
489  Print_N_Log(ERR_MSG,"- Would you like to update it? (Y/N): ");
490 
491  while (updProcRunning)
492  {
493  char FW_Update_entry; // for storing runtime single key entries
494  cin >> FW_Update_entry;
495  cin.ignore();
496  Print_N_Log(LOG_MSG, "%c\n", FW_Update_entry);
497 
498  if(strchr("Yy", FW_Update_entry) != NULL)
499  {
500  while((Ret_FWUpdate = MSP430_FET_FwUpdate(NULL, &MY_MSP430_CallbackFunc, myHandle)) != STATUS_OK) // if update fails
501  {
502  Print_N_Log(ERR_MSG,"* Update failed. (R)etry/(C)ancel? "); // prompt user
503 
504  while(1)
505  {
506  cin >> FW_Update_entry;
507  cin.ignore();
508  Print_N_Log(LOG_MSG, "%c\n", FW_Update_entry);
509  if (strchr("CcRr", FW_Update_entry) != NULL) break;
510  }
511 
512  if (strchr("Cc", FW_Update_entry) != NULL) break;
513  else if (strchr("Rr", FW_Update_entry) != NULL) continue;
514  }
515  if((lVersion == V2_TO_V3_UPGRADE_REQUIRED) && (Ret_FWUpdate == STATUS_OK)) // UIF is now enumerated as CDC device, so reinitialize
516  {
517  Print_N_Log(REPORT_MSG,"* FET has been updated to DLLv3. Reinitializing...\n");
518  Sleep(10000);
519  }
520  Ret_Initialize = MSP430_Initialize(sOpt->pcInterface, &lVersion);
521  uptodate = true;
522  }
523  else if(strchr("Nn", FW_Update_entry) != NULL) break;
524  else continue;
525 
526  // in case of update failure or unsuccessful initialization, quit flasher
527  if ((Ret_FWUpdate == STATUS_ERROR) || (Ret_Initialize == STATUS_ERROR))
528  {
529  inst_state |= ALT_EXIT;
530  StdUseCase_Exit(EXIT_FET_FWUPDATE, 0);
531  }
532  // otherwise continue
533  else
534  {
535  updProcRunning = false;
536  }
537  }
538  }
539  }
540  else
541  {
542  Ret_FWUpdate = STATUS_OK;
543  }
544 
545  if(uptodate)
546  {
547  Print_N_Log(STD_MSG,"* FET firmware is up to date.\n");
548  }
549  else
550  {
551  Print_N_Log(STD_MSG,"* Warning: FW mismatch! Correct functionality not guaranteed!\n");
552  }
553 
554  // abort execution here if NO_TARGET option is selected
555  if(sOpt->pcDeviceName == "NO_TARGET")
556  {
557  MSP430_Close(0);
558  inst_state |= ALT_EXIT;
559  StdUseCase_Exit(EXIT_NO_ERROR, 0);
560  }
561 }

1299  }
1300  while(BPLocal[tempcount] != 0)

1304  }

1320  Addrcount++;

1322  }
1323  BPcount -= invalid;
1324  if(BPflag)
1325  {
1326  for(localcount = 0; BPLocal[localcount] != 0; localcount++)
1327  {
1328  AddrArray[localcount] = BPLocal[localcount];
1329  }
