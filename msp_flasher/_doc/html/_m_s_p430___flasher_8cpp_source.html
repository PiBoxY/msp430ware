75 int main(int argc, char* argv[])
76 {
77  int RetState = 0;
78 
79  // Prepare default options
80  g_sOpt.pcInterface = g_Def.defInterface;
81  g_sOpt.pcEraseType = g_Def.defEraseType;
82  g_sOpt.pcVerify = g_Def.defFALSE;
83  g_sOpt.pcVCC = 0;
84  g_sOpt.pcPassword = g_Def.defPassword;
85  g_sOpt.pcTimeout = g_Def.defTimeout;
86 
87  // Execute Flasher routine
88  RetState = Flasher(argc, argv, &g_sOpt, &g_RTE);
89 
90  switch(RetState)
91  {
92  case 1:
93  Help();
94  return RetState;
95  break;
96  case 2:
97  Exit_Specs();
98  return RetState;
99  break;
100  }
101 
102  StdUseCase_Close(&g_sOpt);
103  return RetState;
104 }

106 //***************************************************************************//
115 int Flasher(int argc, char* argv[], sDefOptions_t* sOpt, sRTE_t* RTE)
116 {
117  int RetState = 0;
118 
119 //==ENABLE LOG===============================================================//
120 
121  EnableLog(argc, argv);
122 
123 //==PRINT VERSION INFO=======================================================//
124 
125  PrintVersion();
126 
127 //==EVALUATE STANDARD OPTIONS================================================//
128 
129  RetState = EvaluateOptions(argc, argv, sOpt, RTE);
130 
131 //==INITIALIZE INTERFACE=====================================================//
132 
133  // std options are evaluated, continue initialization
134  if(!RetState)
135  {
136  StdUseCase_InterfaceInit(&g_sOpt);
137 
138 //==CHECK FIRMWARE COMPATIBILITY=============================================//
139 
140  StdUseCase_CheckFirmware(&g_sOpt);
141 
142 //==READ FW and HW VERSIONS==================================================//
143 
144  StdUseCase_ReadFWVersion();
145  StdUseCase_ReadHWVersion();
146 
147 //==POWER UP ===============================================================//
148 
149  StdUseCase_PowerUp(&g_sOpt);
150 
151 //==GET DEVICE UNDER JTAG CONTROL============================================//
152 
153  StdUseCase_SpecialErase(&g_sOpt);
154  StdUseCase_GetDevice(&g_sOpt);
155 
156  // if Breakpoints should be set
157  if(sOpt->pcBreakpoints != NULL)
158  {
159  StdUseCase_EemInit();
160  }
161 
162 //==ERASE MEMORY (optional)==================================================//
163 
164  StdUseCase_Erase(&g_sOpt);
165 
166 //==PROGRAM & VERIFY / READ OUT DEVICE MEMORY================================//
167 
168  StdUseCase_ReadWrite(&g_sOpt, &g_RTE);
169 
170 //==RESET DEVICE=============================================================//
171 
172  StdUseCase_Reset(PUC_RESET, 0, 0);
173 
174 //==SET BREAKPOINTS==========================================================//
175 
176  // breakpoints are only set if -d switch is used
177  if(sOpt->pcBreakpoints != NULL)
178  {
179  StdUseCase_SetBreakpoints(BPAddr);
180  }
181 
182 //==CHECK IF DEVICE SHOULD BE SECURED========================================//
183 
184  // target device is only secured if -f switch is used
185  StdUseCase_BlowFuse();
186 
187 //==DISPLAY INITIALIZATION RESULTS===========================================//
188 
189  StdUseCase_DisplayResults(argc, argv, &g_sOpt);
190 
191 //==RUN TO BREAKPOINT(S)=====================================================//
192 
193  StdUseCase_RunProgram(&g_sOpt, &BPAddr[0]);
194  }
195 
196 //===========================================================================//
197 
198  return (RetState);
199 }
